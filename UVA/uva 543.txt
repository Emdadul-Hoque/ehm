#include<bits/stdc++.h>
using namespace std;

#define size_n 1000000
#define size_p 1000000

bool flag [size_n+5];
int primes [size_p+5];

void sieve()
{
    int i, j, total=0, value;

    for(i=2; i<=size_n; i++) flag[i]=1;

    value=sqrt(size_n)+1;

    for(i=2; i<value; i++){
        if(flag[i]){
            for(j=i; j*i<=size_n; j++){
                flag[i*j]=0;
            }
        }
    }
}

int main()
{
    int n, i;

    sieve();

    while(scanf("%d",&n)==1){

        if(n==0) return 0;

        for(i=2; i<=n/2; i++){
            if(flag[i] && flag[n-i]){
                printf("%d = %d + %d\n",n,i,n-i);
                break;
            }
        }

        if(i>(n/2)) printf("Goldbach's conjecture is wrong.\n");
    }
    return 0;
}
